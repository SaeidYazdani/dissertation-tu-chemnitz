    4              & WRITE            & 0x81           & Write to a single register                                         & FPGA            \\

    6              & RESET\_FPGA      & 0xFG           & Forces the FPGA to reset itself                                    & FPGA            \\
    \hline
    7              & VI               & 0xF9           & Change V/I Range and gain                                          & GPIO to ADC     \\
    \hline
    8              & SUPPLY           & 0xE1           & Change switching supply output                                     & SPI to PSUs     \\
    \hline
    9              & LIMITER          & 0x8F           & Change limiter resistor value                                      & SPI to Limiter  \\
    \hline
    10             & PAMP             & 0xCF           & Change pre-amp setting                                             & SPI to Amp      \\
    \hline
    11             & PINRELAY         & 0xCA           & Open/Close output relay                                            & GPIO to ADC     \\
    \hline
    12             & READ\_DPRAM      & 0xE5           & Initiate waveform read procedure                                   & FPGA            \\
    \hline
    13             & TRANS\_DPRAM     & 0xE7           & Read from waveform memory                                          & FPGA            \\
    \hline
    14             & SEQ\_DATA        & 0xDA           & Initiate communication with sequencer memory                       & FPGA            \\
    \hline
    15             & SEQ\_TRANSFER    & 0xCE           & Transfer data to sequencer memory                                  & FPGA            \\
    \hline
    16             & SET\_PSU\_SWITCH & 0x32           & Change Analog power supply (old)                                   & GPIO to PSU     \\
    \hline
    17             & AMP\_CTRL        & 0x83           & Configure Amplifier control line                                   & GPIO to Amp     \\
    \hline
    18             & WRITE\_MULTI     & 0xFC           & Write to multiple registers                                        & FPGA            \\
    \hline
    19             & READ\_MULTI      & 0xFA           & Read from multiple registers                                       & FPGA            \\
    \hline
    20             & CHANGE\_RANGE    & 0xF8           & Set multiple subsystems for total change or range and control mode & All boards \\
    \hline
    21             & CONTROL\_MODE    & 0xF6           & Switch between Voltage or Current source mode                      & FPGA            \\
    \hline
    22             & QUERY\_STATE     & 0x92           & Reads current configuration of the system                          & All boards      \\
    \hline
    23             & READ\_RAM\_INIT  & 0xA2           & Initialize reading from a specific internal memory of the FPGA & FPGA \\

    \hline
    25             & INIT\_TEST       & 0xA3           & Initialize a given test routine                                    & FPGA            \\
    \hline
    26             & TRIG\_TEST       & 0xA4           & Starts a previously initialized test routine                       & Systemwide      \\
